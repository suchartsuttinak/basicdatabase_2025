โดยยึดหลักดังนี้

1) ใช้โครงสร้าง UI แบบ modern / clean / international standard
2) ต้อง responsive ทุกขนาดหน้าจอ
3) มือถือและจอเล็กต้องใช้งานได้จริง และต้องมี scroll ใน modal/body ถ้าจำเป็น
4) ป้องกันปัญหาเดิมที่เคยเจอ:
   - modal จอเล็กแล้วไม่มี scroll
   - ฟอร์มชิดซ้ายหรือผิดสัดส่วน
   - ปุ่มไม่เรียงในแถวเดียว
   - ปุ่มซ้อนกันหรือขนาดไม่เท่ากัน
   - radio / checkbox card ตัดบรรทัดไม่สวย
   - table ล้นจอโดยไม่มี scroll
   - create และ edit หน้าตาไม่เหมือนกัน
   - CSS ชนกัน หรือ selector ทับกันมั่ว
5) ให้ใช้โครงสร้าง create/edit มาตรฐานเดียวกัน
6) ถ้ามี modal:
   - modal-content ต้องจัด layout ให้ถูก
   - form ภายใน modal ต้องเป็น flex column
   - modal-body ต้อง scroll ได้บนจอเล็ก
   - footer ต้องไม่ลอยทับเนื้อหา
7) ถ้ามี action buttons:
   - desktop ให้อยู่แถวเดียวกัน
   - mobile ถ้าพื้นที่ไม่พอให้ scroll แนวนอนหรือเรียงลงอย่างสวยงามตามความเหมาะสม
8) ถ้ามี radio-card หรือ option-card:
   - ความกว้างสมดุล
   - ความสูงใกล้เคียงกัน
   - ข้อความไม่แตกผิดรูป
9) ขอให้คุณ:
   - วิเคราะห์ปัญหาก่อน
   - ปรับ HTML + CSS + JS ที่จำเป็นให้ครบ
   - ส่งโค้ดที่พร้อมใช้งานได้ทันที
   - อย่าทำให้กระทบหน้าอื่น
   - ถ้าต้อง override CSS ให้ทำอย่างปลอดภัย
10) ช่วยจัดโครงสร้าง class ให้ดูแลง่าย และลดปัญหาซ้ำในอนาคต

3. Mobile ต้องใช้งานได้จริง:
   - modal ต้อง scroll ได้
   - table ต้อง scroll แนวนอน
   - ปุ่มไม่ซ้อน / ไม่ล้น / ไม่แตกบรรทัดมั่ว

   - desktop อยู่แถวเดียว
   - mobile ถ้าไม่พอให้ scroll แนวนอน หรือเรียงอย่างสวย
   - ขนาดเท่ากัน ไม่ยุบ ไม่ล้น

6. form / modal:
   - form ใช้ flex column
   - modal-body scroll ได้
   - footer ไม่ทับเนื้อหา
   - create และ edit ต้องหน้าตาเหมือนกัน

7. radio / checkbox card:
   - ขนาดสมดุล
   - ไม่แตก layout

   - ต้อง scoped เช่น .psychiatric-page / .addictive-page
   - ห้ามกระทบหน้าอื่น
   - ไม่แก้ global เช่น .btn .table โดยตรง
   - ใส่ CSS มาให้พร้อมในหน้านี้เลย

9. โครงสร้าง:
   - จัด class ให้เป็นระบบ อ่านง่าย แก้ไขง่าย
   - reusable ได้ในอนาคต

10. ส่งโค้ด:
   - ส่ง HTML + CSS + JS (ถ้าจำเป็น)
   - พร้อมใช้งาน copy ไปวางได้เลย

use App\Http\Controllers\ClientController;

// 🔹 ทุก role เข้าดูได้
Route::middleware(['auth'])->group(function () {
    Route::get('/client', [ClientController::class, 'index'])->name('client.index');
    Route::get('/client/show/{id}', [ClientController::class, 'show'])->name('client.show');
});

// 🔹 เฉพาะคนที่แก้ไขได้
Route::middleware(['auth', 'role:admin,manager,social_worker'])->group(function () {
    Route::get('/client/create', [ClientController::class, 'create'])->name('client.create');
    Route::post('/client/store', [ClientController::class, 'store'])->name('client.store');
    Route::get('/client/edit/{id}', [ClientController::class, 'edit'])->name('client.edit');
    Route::put('/client/update/{id}', [ClientController::class, 'update'])->name('client.update');
});

// 🔹 เฉพาะ admin ลบได้
Route::middleware(['auth', 'role:admin'])->group(function () {
    Route::delete('/client/delete/{id}', [ClientController::class, 'destroy'])->name('client.delete');
});

งานนี้ให้ “คง route เดิม / ชื่อ route เดิม / view เดิม / โครงระบบเดิม” ไว้ทั้งหมด
ห้ามเปลี่ยน route เว้นแต่ผมสั่ง

เป้าหมาย:

* อุดช่องโหว่การเดา URL เข้า record/client ของคนที่ไม่มีสิทธิ์
* แก้ที่ controller เป็นหลัก
* ใช้แนวทางเดียวกับหน้า EducationRecord
* ถ้าเป็นข้อมูลลูกของ client ให้ตรวจสิทธิ์ผ่าน Client::forUser(auth()->user()) ทุกจุด
* ห้าม refactor จนโค้ดสั้นลงมาก
* ขอแบบ “เต็มไฟล์ วางแทนได้เลย”
* คงชื่่อ method เดิม
* คงลำดับ method เดิมให้มากที่สุด
* คอมเมนต์กำกับจุดที่แก้ให้ด้วย

สิ่งที่ผมจะส่งให้:

1. route เดิม
2. controller เดิม
3. ถ้าจำเป็นจะส่ง model ที่เกี่ยวข้องเพิ่ม

สิ่งที่คุณต้องทำ:

1. วิเคราะห์ก่อนว่าช่องโหว่อยู่ตรงไหน
2. บอกผมสั้น ๆ ก่อนแก้ว่า จะไม่แก้ route ถ้าไม่จำเป็น
3. จากนั้นส่ง controller ฉบับเต็มที่แก้แล้วทั้งไฟล์
4. ถ้ามีจุดที่ต้องแก้เพิ่มนอก controller ให้บอก “เฉพาะที่จำเป็นจริง ๆ”
5. อย่าเปลี่ยนชื่อ route เพราะจะทำให้ของเดิมพัง

เงื่อนไขสำคัญ:

* เน้นแก้สิทธิ์การเข้าถึงเป็นหลัก
* อย่ายุบ helper จนผมเทียบไฟล์เดิมไม่ได้
* อย่าตัดโค้ดเดิมออกถ้าไม่จำเป็น
* ถ้ามี route ซ้ำ ให้เตือนก่อน อย่าเปลี่ยนเองทันที
* ถ้า controller เดิมใช้ findOrFail ตรง ๆ ให้เปลี่ยนเป็นตรวจสิทธิ์ก่อน
* ถ้าเป็น record ลูก ให้เช็กสิทธิ์ผ่าน client_id ของ record นั้นด้วย

รูปแบบคำตอบที่ผมต้องการ:

* สรุปปัญหา 3-7 บรรทัด
* บอกแนวแก้
* ส่งโค้ดเต็มไฟล์พร้อมวางแทนได้เลย

  <?php

namespace App\Console\Commands;

use Illuminate\Console\Command;
use App\Models\User;
use Illuminate\Support\Facades\Hash;

class CreateAdminUser extends Command
{
    protected $signature = 'user:create-admin';
    protected $description = 'Create admin user';

    public function handle()
    {
        $name = $this->ask('ชื่อ');
        $email = $this->ask('อีเมล');
        $password = $this->secret('รหัสผ่าน');

        $user = User::create([
            'name' => $name,
            'email' => $email,
            'password' => Hash::make($password),
            'role' => 'admin',
            'status' => '1',
        ]);

        $this->info('สร้าง admin สำเร็จ: ' . $user->email);
    }
}
